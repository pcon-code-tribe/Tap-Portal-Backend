// Table for Admin
CREATE TABLE ADMIN
(
    RegNo VARCHAR(21) PRIMARY KEY
    Name VARCHAR(255) 
    Email VARCHAR(255) 
    Password VARCHAR(255)
    Branch VARCHAR(255)   
);

//TABLE FOR USER
CREATE TABLE USER
( 
  RegNo VARCHAR(21) PRIMARY KEY
  Name VARCHAR(255)
  Branch VARCHAR(255)
  Email VARCHAR(255)
  Password VARCHAR(255)
  CV VARCHAR(255)
  NoOfPlacements INT
  Cgpa INT  
);

//TABLE FOR USER LOG
CREATE TABLE  USER LOGGER
(
  id Int auto increment 
  User id VARCHAR(255)
  log VARCHAR(255)
  Time stamp DATETIME
  FOREIGN KEY(USER ID) REFERENCES USER(RegNo) ON DELETE CASCADE
);

//TABLE FOR ADMIN LOG
CREATE TABLE  ADMIN LOGGER
(
  id Int auto increment 
  User id VARCHAR(255)
  log VARCHAR(255)
  Time stamp DATETIME
  FOREIGN KEY(USER ID) REFERENCES ADMIN(RegNo) ON DELETE CASCADE
);

//TABLE FOR COMPANY
CREATE TABLE COMPANY
(
    Id VARCHAR(255) PRIMARY KEY
    Date of visit DATE
    Name VARCHAR(255)
    Last date of apply DATE
    Eligible branches VARCHAR(255)
    Package offers INT
    MINIMUM CGPA INT
    Description VARCHAR(500)
    PDF VARCHAR(1000)
);

//TABLE FOR APPLIED COMPANY
poolCREATE TABLE APPLIED COMPANY
(
    COMPANY_ID VARCHAR(255) PRIMARY KEY
    RegNo VARCHAR(21)
    Status VARCHAR(255)
    FOREIGN KEY(COMPANY_ID) REFERENCES COMPANY(Id) ON DELETE CASCADE
    FOREIGN KEY(RegNo) REFERENCES USER(RegNo) ON DELETE CASCADE
);
